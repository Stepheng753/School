	\begin{problem}{1}
		Given the ODE:
		\[y'' - y = 0.\]
		\begin{enumerate}[label = (\alph*)]
			\item Show that $y_1(t) =e^t$ and $y_2(t) = e^{-t}$ are solutions to the differential equation.  In addition, show that this pair form a linearly independent set using the definition of linear independence.
			\\ \\
			We can show that the given functions are solutions to the ODE by substituting them into the ODE and the equality remaining true.  So notice the following:
			\[y_1(t)'' - y_1(t) = \frac{d^2}{dt^2}\left(e^{t}\right) - e^t = e^t - e^t = 0  \]
			\[y_2(t)'' - y_2(t) = \frac{d^2}{dt^2}\left(e^{-t}\right) - e^{-t} = e^{-t} - e^{-t} = 0\]
			\\ 
			\textbf{\boldmath Thus we have shown that $y_1(t) = e^t$ and $y_2(t) = e^{-t}$ are solutions to the given ODE.}
			\\ \\
			\begin{proof}
				Let the following be true:
				\[c_1e^{t} + c_2e^{-t} = 0\]
				where $c_1$ and $c_2$ are constants. We can substitute $t = 0$ and $t = 1$ to get the following:
				\[c_1 + c_2 = 0 \qquad c_1e + c_2e^{-1} = 0\]
				From the left equation, we get that $c_1 = -c_2$.  By substituting this into the right equation we get:
				\[-c_2e + c_2e^{-1} = 0\]
				Solving this we get that $c_2e^{-1} = c_2e$.  Thus the only way for this to be true is if $c_1 = c_2 = 0$.
				\\ \\ 
				\textbf{Thus proving the set to be linear independent}. \\ 
			\end{proof}   
			\newpage
			\item Also,  show  that $y_1(t) = \sinh\,t$ and $y_2(t) = \sinh(1-t)$  are  solutions  to  the  differential equation.  In addition, show that this pair form another linearly independent set.
			\\ \\
			We can show that the given functions are solutions to the ODE by substituting them into the ODE and the equality remaining true.  So notice the following:
			\[y_1(t)'' - y_1(t) = \frac{d^2}{dt^2}\left(\sinh\,t\right) - \sinh\,t = \sinh\,t - \sinh\,t = 0\]
			\[y_2(t)'' - y_2(t) = \frac{d^2}{dt^2}\left(\sinh(1-t)\right) - \sinh(1-t) = \sinh(1-t) - \sinh(1-t) = 0\]
			\\
			\textbf{\boldmath Thus we have shown that $y_1(t) = \sinh(t)$ and $y_2(t) = \sinh(1-t)$ are solutions to the given ODE.}
			\\ \\
			\begin{proof}
				Let the following be true:
				\[c_1\sinh(t) + c_2\sinh(1-t) = 0\]
				where $c_1$ and $c_2$ are constants.  We can substitute $t = 0$ and $t = 1$ to get the following:
				\[c_2\sinh(1) = 0 \qquad c_1\sinh(1) = 0\]
				From this, we get that $c_1 = c_2 = 0$. \\ \\
				\textbf{Thus proving the set to be linear independent}. \\
			\end{proof}  
		\end{enumerate}
	\end{problem}

	\begin{problem}{2}
		Consider the following second order linear homogeneous differential equation:
		\[y'' - 2ay' + (a^2 + b^2)y = 0,\]
		where the parameters are fixed and positive, so assume $a > 0$ and $b > 0$.
		\begin{enumerate}[label = (\alph*)]
			\item Find the general solution to this ordinary differential equation (ODE).
			\\ \\
			Notice the characteristic equation:
			\[\lambda^2 - 2a\lambda + (a^2 + b^2) = 0\]
			Using the quadratic formula, we can solve for $\lambda$:
			\[\lambda = \frac{2a \pm \sqrt{4a^2 - 4(a^2 + b^2)}}{2} = \frac{2a \pm \sqrt{-4b^2}}{2} = a \pm bi\]
			Because we have $\lambda \in \mathbb{C}$, we get the following general solution:
			\[y(t) = Ce^{(a \pm bi)t} = Ce^{at}e^{(\pm i\,bt)} = e^{at} \left(c_1\cos\,bt + c_2\,i\sin\,bt\right)\]
			\item Find the unique solution to the initial value problem (IVP), where the initial conditions for the ODE are:
			\[y(0) = y_0 \qquad \text{and} \qquad y'(0) = z_0.\]
			Notice the following:
			\[y(0) = c_1 = y_0\]
			\[y'(t) = e^{at}\left[\left(c_1a\cos\,bt + c_2a\,i\sin\,bt\right) + \left(-c_1b\sin\,bt + c_2b\,i\cos\,bt\right)\right]  \]
			\[y'(0) = c_1a + c_2bi = y_0a + c_2bi = z_0 \qquad c_2 = \frac{z_0 - y_0a}{bi}\]
			Thus we get the following unique solution:
			\[y(t) = e^{at} \left(y_0\cos\,bt + \frac{z_0 - y_0a}{b}\sin\,bt\right)\]
			\newpage
			\item Now consider the ODE with boundary conditions:
			\[y(0) = A \qquad \text{and} \qquad y(x_0) = B.\]
			Give conditions on the boundary condition parameters, $A, B, \text{ and } x_0 > 0$, such that this boundary value problem (BVP) has:
			\\ \\
			(i) A unique solution \qquad (ii) No solution \qquad (iii) Infinitely many solutions
			\\ \\
			When this BVP has a unique solution, give the solution to the BVP.
			\begin{enumerate}[label = (\roman*)]
				\item Notice the following:
				\[y(0) = c_1 = A\]
				\[y(x_0) = e^{ax_0} \left(A\cos\,bx_0 + c_2\,i\sin\,bx_0\right) = B\]
				\[c_2 = \frac{Be^{-ax_0} - A\cos\,bx_0}{i\,\sin\,bx_0} \]
				Thus we get the unique solution:
				\[y(t) = e^{at} \left(A\cos\,bt + \frac{Be^{-ax_0} - A\cos\,bx_0}{\sin\,bx_0}\sin\,bt\right)\]
				\item Let $x_0 = c \pi / b$ with $c$ being an even positive integer. Also let $B \not = Ae^{ax_0}$.  Now notice the following:
				\[y(0) = c_1 = A\]
				\[y(x_0) = e^{ax_0}c_1 = B \qquad c_1 = \frac{B}{e^{ax_0}}\]
				Because we have a conflict in the value of $c_1$, thus we get no solution.
				\item Let $x_0 = c \pi / b$ with $c$ being an even positive integer. Also let $B = Ae^{ax_0}$.  Now notice the following:
				\[y(0) = c_1 = A\]
				\[y(x_0) = e^{ax_0}c_1 = 0 \qquad c_1 = A\]
				Thus we get infinite solutions.
			\end{enumerate}
		\end{enumerate}
	\end{problem}
