			$$
			y'' + 16y = 12\sec^2(4x)
			$$

			$$
			\lambda^2 + 16 = 0 \\
			$$

			$$
			y_1 = \cos(4x), \qquad y_2 = \sin(4x)

			W_{[y_1,y_2]} = 
			\begin{vmatrix}
				\cos(4x) & \sin(4x) \\
				-4\sin(4x) & 4\cos(4x)
			\end{vmatrix}

			$$
		Now we can find the particular solution:
			\begin{align*}
				y_p &= -\cos(4x) \int^{x} \frac{12\sin(4s)\sec^2(4s)}{4}\,ds + \sin(4x)\int^{x} \frac{12\cos(4s)\sec^2(4s)}{4}\,ds \\

 			\end{align*}

			$$

			$$

				rref
